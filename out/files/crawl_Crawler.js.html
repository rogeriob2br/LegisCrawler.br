const htmlparser = require('htmlparser2');
const request = require('request-promise-native');
const chalk = require('chalk');
const debug = require('debug')('crawler');
const Type = require('../models/Type');
const error = require('../helpers/error');
const ErrorFixer = require('./ErrorFixer');

/**
 * Crawl urls for the desired content
 * @module Crawler
 * @class Crawler
 */
class Crawler {
  /**
   * Crawler constructor sets the type object
   * @method constructor
   * @constructor
   */
  static constructor() {
    this.type = {
      LEGISLATION: 0,
      DATE: 1,
    };
  }

  /**
   * Crawl Planalto's main page: http://www4.planalto.gov.br/legislacao/portal-legis/legislacao-1
   * Capture all links and return the link and the title as a promise
   * @method mainLegislationPage
   * @static
   * @return {Promise} Array of legislations with name and link
   */
  static mainLegislationPage() {
    let processing = false;
    let captureText = false;
    let ignoreCount = 0;
    const legislations = [];

    return new Promise((resolve, reject) => {
      const requestoptions = {
        url: 'http://www4.planalto.gov.br/legislacao/portal-legis/legislacao-1',
      // encoding: 'latin1',
      };
      request(requestoptions)
        .then((html) => {
          let url = '';
          const parser = new htmlparser.Parser({
            onopentag(tag, attribs) {
              // Check if the page processing is activated
              if (processing === true) {
                // add divs inside the capture div to the ignore counter
                if (tag === 'div') {
                  ignoreCount += 1;
                } else if (tag === 'a') {
                  url = attribs.href;
                  captureText = true;
                }
              // Check if the 'content-core' id is found to activate the processing
              } else if (tag === 'div' && attribs.id === 'content-core') {
                // debug(chalk.green(`found tag ${tag} with id ${attribs.id}`));
                processing = true;
              }
            },
            ontext(name) {
              if (processing === true && captureText === true) {
                // debug(chalk.blue(`captured ${name}`));
                url = ErrorFixer.fixCrawlerMainLegislationPage(name, url);
                if (url !== null) {
                  legislations.push({
                    name,
                    url,
                  });
                }
                captureText = false;
              }
            },
            onclosetag(tag) {
              if (tag === 'div' && processing === true) {
                if (ignoreCount === 0) {
                  // debug(chalk.green('finished processing'));
                  processing = false;
                } else if (ignoreCount > 0) {
                  // debug(chalk.yellow(`removed ignore div # ${ignoreCount}`));
                  ignoreCount -= 1;
                }
              }
            },
          }, {
            decodeEntities: true,
          });
          parser.write(html);
          parser.end();

          // debug(chalk.green('scrapedContent', legislations));
          resolve(legislations);
        })
        .catch((err) => {
          error('InitialPage', 'Could not scrap page', err);
          reject(error);
        });
    });
  }

  static checkType(url) {
    return new Promise((resolve, reject) => {
      // Check if url is an HTML
      const urlSplit = url.split('/');
      debug(urlSplit);
      const htmlCheckSplit = urlSplit[urlSplit.length - 1].split('.');
      debug(htmlCheckSplit);
      const htmlCheck = htmlCheckSplit[htmlCheckSplit.length - 1].toLowerCase();

      if (htmlCheck === 'htm' || htmlCheck === 'html') {
        resolve(Type.LEGISLATION);
      }

      // Check the other types
      let processing = false;
      let ignoreCount = 0;
      const requestoptions = {
        url,
        encoding: 'latin1',
      };
      request(requestoptions)
        .then((html) => {
          const parser = new htmlparser.Parser({
            onopentag(tag, attribs) {
              // Check if the page processing is activated
              if (processing === true) {
                // add divs inside the capture div to the ignore counter
                if (tag === 'div') {
                  ignoreCount += 1;
                } else if (tag === 'a') {
                  url = attribs.href;
                  captureText = true;
                }
              // Check if the 'content-core' id is found to activate the processing
              } else if (tag === 'div' && attribs.id === 'content-core') {
                // debug(chalk.green(`found tag ${tag} with id ${attribs.id}`));
                processing = true;
              }
            },
            ontext(name) {
              if (processing === true) {
                debug(chalk.blue(`captured ${name}`));
              }
            },
            onclosetag(tag) {
              if (tag === 'div' && processing === true) {
                if (ignoreCount === 0) {
                  // debug(chalk.green('finished processing'));
                  processing = false;
                } else if (ignoreCount > 0) {
                  // debug(chalk.yellow(`removed ignore div # ${ignoreCount}`));
                  ignoreCount -= 1;
                }
              }
            },
          }, {
            decodeEntities: true,
          });
          parser.write(html);
          parser.end();

          resolve(Type.DATE);
        })
        .catch((err) => {
          error('checkType', `Could not check type of ${url}`, err);
          reject(error);
        });
    });
  }
}

module.exports = Crawler;
